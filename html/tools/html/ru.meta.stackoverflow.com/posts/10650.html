int multp(int* f, int* m, int* r) {
    for (usi i = 1; i <= f[0]; i++) {
        for (usi j = 1; j <= m[0]; j++) {
            r[i + j - 1] += f[i] * m[j];
        }
    }
    for (usi i = 1; i < r[0]; i++) {
        r[i + 1] += (r[i] / 10);
        r[i] %= 10;
    }
    while(r[r[0] - 1] == 0) {
        r[0]--;
    }
    return r[0];
}//write the max size of 'r' array in r[0]
