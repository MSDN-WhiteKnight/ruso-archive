<pre class="lang-cpp prettyprint-override"><code>std::atomic&lt;bool&gt; should_stop=false;

{
    mouse_event((0x0001), 1, 4, 0, 0);
    Sleep(100);    
    if(should_stop) return;
    mouse_event((0x0001), 1, 4, 0, 0);
    Sleep(100);
    if(should_stop) return;
    //...    
}

//...

should_stop = true; //остановить поток
