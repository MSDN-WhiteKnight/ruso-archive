        wchar_t file_path[1024];
        int const items_count_to_scan = 1;
        if(items_count_to_scan == wscanf_s(L"%s", file_path, _countof(file_path)))
        {
            wprintf_s(L"got file path \"%s\"\n", file_path);
        }
    }
    _wsystem(L"pause");
    return 0;
}
