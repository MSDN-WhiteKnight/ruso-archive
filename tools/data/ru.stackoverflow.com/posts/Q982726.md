_LoadLibraryA LoadLibraryAPointer;

for (int i = 0; i &lt; pIED-&gt;AddressOfFunctions; i++)
{
    if (!hashComparatorA(MESSAGE_BOX_HASH, (char*)user32 + Name[i]))

        break;
    }
}
