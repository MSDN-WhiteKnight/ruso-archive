<pre><code>double Root(double base, int degree)
{
  if (base &lt; 0 &amp;&amp; degree % 2 == 1)
    return -Math.Pow(Math.Abs(base), 1.0 / degree);
  else 
    return Math.Pow(base, 1.0 / degree);
}
