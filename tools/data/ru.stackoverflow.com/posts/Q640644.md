<pre><code>int[][] a2 = new int[][]{ new int[]{ 1, 2 }, new int[] { 2, 3 }, new int[] { 3, 4 } };
for (int i = 0; i &lt; a2.Length; i++)
{
    for (int j = 0; j &lt; a2[i].Length; j++)
