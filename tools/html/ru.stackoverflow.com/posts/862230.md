<p>Нужно переопределить метод IsReadOnly в ConfigurationSection:</p>

<pre><code>public class CustomConfigurationSection : ConfigurationSection
{

    public override bool IsReadOnly()
    {
        return false;
    }

    //...
}
</code></pre>
