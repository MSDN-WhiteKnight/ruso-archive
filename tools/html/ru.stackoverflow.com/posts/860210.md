<p>Если нужно временно скрыть или отобразить консоль, можно использовать следующие функции:</p>

<pre><code>#include &lt;Windows.h&gt;

void HideConsole()
{
    ::ShowWindow(::GetConsoleWindow(), SW_HIDE);
}

void ShowConsole()
{
    ::ShowWindow(::GetConsoleWindow(), SW_SHOW);
}    
</code></pre>
