<pre><code>using System.Runtime.InteropServices;
using System.Runtime.InteropServices.ComTypes;

[DllImport("mylib.Dll")]
static extern int EvalCRC(
    IStream data, 
    IStream parms, 
    IStream output
);
</code></pre>
